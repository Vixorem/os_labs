#include <stdio.h>
#include <sys/time.h>

void check_xor(char *fileName, int expectedSum) {
    unsigned char sum = 0; 
    char buff; 
    FILE *file = fopen(fileName, "r");

    while (fread(&buff, 1, 1, file) != 0) { 
        sum ^= buff;
    }

    fclose(file);

    printf("Ожидаемая сумма %d\n", expectedSum);
    printf("Полученная сумма %d\n", sum);
}

int main() {
    check_xor("../tests/1.txt", 96);

    return 0;
}
